// --- 1. CONFIGURATION (अपनी स्ट्रीम और API की डिटेल्स यहाँ डालें) --- const STREAM_URL = "https://stream.zeno.fm/f9t7t888w0euv"; const API_METADATA_URL = "https://api.zeno.fm/mounts/metadata/subscribe/f9t7t888w0euv"; // उदाहरण के लिए Zeno API // DOM Elements const player = document.getElementById('radio-player'); const playBtn = document.getElementById('play-btn'); const wavesContainer = document.getElementById('waves'); const currentShow = document.getElementById('current-show'); const currentRj = document.getElementById('current-rj'); let audioContext, analyser, source, animationFrameId; // --- 2. REAL AUDIO VISUALIZER (असली म्यूज़िक बीट्स पर चलने वाली वेव्स) --- function setupVisualizer() { if (audioContext) return; // दोबारा सेटअप न हो // Web Audio API सेटअप audioContext = new (window.AudioContext || window.webkitAudioContext)(); analyser = audioContext.createAnalyser(); // प्लेयर को ऑडियो सोर्स से जोड़ें source = audioContext.createMediaElementSource(player); source.connect(analyser); analyser.connect(audioContext.destination); analyser.fftSize = 32; // वेव्स की संख्या कंट्रोल करने के लिए const bufferLength = analyser.frequencyBinCount; const dataArray = new Uint8Array(bufferLength); // वेव्स को HTML में जनरेट करना wavesContainer.innerHTML = ''; for (let i = 0; i < bufferLength; i++) { const bar = document.createElement('div'); bar.className = 'bar'; wavesContainer.appendChild(bar); } const bars = document.querySelectorAll('.bar'); // वेव्स को म्यूज़िक के साथ एनिमेट करने का फंक्शन function animateWaves() { if (player.paused) return; animationFrameId = requestAnimationFrame(animateWaves); analyser.getByteFrequencyData(dataArray); bars.forEach((bar, index) => { // ऑडियो फ्रीक्वेंसी के आधार पर हाइट सेट करना const height = (dataArray[index] / 255) * 60; bar.style.height = `${Math.max(5, height)}px`; }); } animateWaves(); } // --- 3. PLAY / PAUSE CONTROL --- function togglePlay() { // ब्राउज़र की सुरक्षा नीति के कारण AudioContext यूज़र क्लिक पर ही शुरू होना चाहिए if (audioContext && audioContext.state === 'suspended') { audioContext.resume(); } if (player.paused) { player.src = STREAM_URL; // बफरिंग कम करने के लिए क्लिक पर सोर्स लोड करें player.play() .then(() => { playBtn.innerText = "PAUSE LIVE"; wavesContainer.style.visibility = 'visible'; setupVisualizer(); fetchLiveMetadata(); // लाइव डेटा लाना शुरू करें }) .catch(err => { console.error("Playback error:", err); alert("रेडियो स्ट्रीम लोड नहीं हो सकी। कृपया दोबारा कोशिश करें।"); }); } else { player.pause(); player.src = ""; // बैकग्राउंड में डेटा वेस्ट होने से रोकने के लिए playBtn.innerText = "PLAY LIVE"; wavesContainer.style.visibility = 'hidden'; cancelAnimationFrame(animationFrameId); } } // --- 4. LIVE METADATA FETCHING (सर्वर से लाइव गाने/RJ का नाम लाना) --- function fetchLiveMetadata() { // प्रीमियम वेबसाइट्स Server-Sent Events (SSE) या Fetch API का यूज़ करती हैं if (!!window.EventSource && API_METADATA_URL) { const source = new EventSource(API_METADATA_URL); source.onmessage = function(event) { const data = JSON.parse(event.data); if (data.streamTitle) { // मान लेते हैं सर्ver से "Show Name - RJ Name" आ रहा है const meta = data.streamTitle.split(" - "); currentShow.innerText = `शो: ${meta[0] || 'डमरू बीट्स'}`; currentRj.innerText = `आरजे: ${meta[1] || 'लाइव'}`; } }; source.onerror = function() { source.close(); // अगर API फेल हो जाए तो टाइम-बेस्ड बैकअप लॉजिक चलाएं runBackupSchedule(); }; } else { runBackupSchedule(); } } // बैकअप लॉजिक (अगर सर्वर API काम न करे) function runBackupSchedule() { const hour = new Date().getHours(); if (hour >= 7 && hour < 12) { currentShow.innerText = "शो: मॉर्निग मंत्रा"; currentRj.innerText = "आरजे: अमित"; } else if (hour >= 12 && hour < 18) { currentShow.innerText = "शो: डमरू बीट्स"; currentRj.innerText = "आरजे: राहुल"; } else { currentShow.innerText = "शो: महफ़िल-ए-ग़ज़ल"; currentRj.innerText = "आरजे: नेहा"; } } // --- 5. PREMIUM COMMENT SYSTEM (LOCAL STORAGE & SANITIZATION) --- // इससे पेज रीफ्रेश होने पर भी पुराने कमेंट्स गायब नहीं होंगे document.addEventListener("DOMContentLoaded", () => { loadComments(); runBackupSchedule(); // पेज लोड होते ही शुरुआती शो डिटेल्स दिखाने के लिए }); function addComment(e) { e.preventDefault(); const usernameInput = document.getElementById('username'); const commentTextInput = document.getElementById('comment-text'); // सुरक्षा के लिए XSS (Hacking) प्रोटेक्शन (HTML टैग्स हटाना) const username = usernameInput.value.replace(//g, ">"); const commentText = commentTextInput.value.replace(//g, ">"); if(!username || !commentText) return; const commentData = { username, text: commentText, time: new Date().toLocaleTimeString() }; // लोकल स्टोरेज में सेव करना let comments = localStorage.getItem('radio_comments') ? JSON.parse(localStorage.getItem('radio_comments')) : []; comments.unshift(commentData); // नए कमेंट को सबसे ऊपर रखें localStorage.setItem('radio_comments', JSON.stringify(comments.slice(0, 50))); // सिर्फ आखिरी 50 कमेंट्स रखें renderComments(); // फॉर्म खाली करना commentTextInput.value = ''; } function loadComments() { renderComments(); } function renderComments() { const list = document.getElementById('comments-list'); if(!list) return; let comments = localStorage.getItem('radio_comments') ? JSON.parse(localStorage.getItem('radio_comments')) : []; list.innerHTML = comments.map(c => `
${c.username} ${c.time}

${c.text}

`).join(''); }
